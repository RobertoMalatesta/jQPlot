html tag & Call jQPlot building system

<script> 
$(function(){
  $.jqplot("chart", [[1, 2, 3]]);
});
</script>

  });

With plugins

<div id="chart"></div>
<script> 
$(function(){
  var s1 = [['/home',46], ['/var',14], ['/tmp',5], ['/',35]];

  var plot1 = $.jqplot('chart', [s1], {
    title: 'Disc usage',
    seriesDefaults: {
      renderer: $.jqplot.PieRenderer,
      rendererOptions: {
        showDataLabels: true,
        labelColor: "red",
        showDataCategoryOnHover : true,
        sliceMargin: 1,
        ringMargin: 8,
      }
    },
    legend: { show:true, location: 'e'}
  });
});
</script>
